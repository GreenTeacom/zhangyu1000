	\begin{ti}

				f(x) = \sum_{n=0}^{\infty} a_{n} x^{n},
			\]

				f(x) = \sum_{n=1}^{\infty} a_{2n-1} x^{2n-1};

			\item 求 $I = \int_{0}^{1} \frac{1}{2 - x} \ln \frac{1}{x} \dd{x}$.
		\end{enumerate}
